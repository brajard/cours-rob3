\begin{frame}
\frametitle{Que fait l'ordinateur ?}
\begin{itemize}

\item Un ordinateur traite de façon automatique de l'information
\begin{itemize}
\item Sous forme magnétique ou électrique,\\
\item Représentation binaire (i.e. 0 et 1).\\
\end{itemize}

\item Besoin d'abstraction pour représentaiton l'information utilement.
\begin{itemize}
\item Entiers,\\
\item Réels,\\
\item Caractères,\\
\item Mots.\\
\end{itemize}

\item Besoin d'abstraction pour représenter les traitements
\begin{itemize}
\item Affectations,
\item Boucles,
\item Opération arithmétiques,
\item Tests.
\end{itemize}

\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Historique}
\begin{description}
\item[1969 : ] Création du système UNIX (Lab. Bell, K. Thompson et D. Ritchie). UNIX est intialement écrit en assembleur.\\
\item[1970 : ] K. Thompson définit le langage B et réécrit UNIX dans ce langage.\\
\item[1972 : ] D. Ritchie définit le langage C.\\
\item[1973 : ] UNIX (version 5) est réécrit en C à 90\%\\
\item[1989 : ] l'ANSI (American National Standard Institute) définit une norme du langage C : C ANSI.\\ 
\end{description}
\end{frame}

\begin{frame}
\frametitle{Caractéristiques du C}
\begin{itemize}
\item Langage de programmation impérative.\\
\item Programmation modulaire.\\
\item Langage de bas-niveau (proche du sytème).\\
\item Simplicité et minimalité du langage.\\
\item Bibliothèques standards assez riches.\\
\item Exécutable généré rapide et de petite taille.\\
\item Langage permissif : faiblement typé (le compilateur de vérifie pas tout).\\
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Création d'un programme}
\begin{enumerate}

\item Écriture d'un programme dans un éditeur de texte (emacs, gedit, ...)
\begin{itemize}
\item Sauvegarde dans un fichier : \verb|fichier_source.c|
\end{itemize}

\item Compilation sous Linux avec \verb|gcc| (dans un terminal)
\begin{itemize}
\item \verb|gcc -Wall -o fichier_executable fichier_source.c|
\end{itemize}

\item Exécution comme une commande (dans un terminal)
\begin{itemize}
\item \Verb[commandchars=\\\{\}]|./fichier_executable\textit{ [paramètres]}|
\end{itemize}

\end{enumerate}
\end{frame}

\begin{frame}[fragile]
\frametitle{Indentation}
\begin{itemize}
\item La façon de présenter un programme n'a pas d'impact sur ses performances.\\
\item Mais, indenter un programme le rend plus lisible et donc plus facile à corriger et donc à faire fonctionner.\\
\item Indenter : décaler de quelques espaces les parties du code dans un bloc.\\
\end{itemize}
\begin{columns}
\column{.47\textwidth}
\begin{codeblock}{Avec indentation}
\lstset{escapeinside={§§}}
\begin{codeC}
#include <stdio.h>

/*Programme principal*/
int main()
{
  printf("Hello world");
  return(0);
}
\end{codeC}
\end{codeblock}

\column{.47\textwidth}
\begin{codeblock}{Sans indentation}
\lstset{escapeinside={§§}}
\begin{codeC}

gramme principal*/int 
main(){printf("Hello 
world");return(0);}
\end{codeC}
\end{codeblock}

\end{columns}
