\begin{frame}
\frametitle{Quelques définitions}

\begin{block}{Structure récursive}
Structure de données dans laquelle un membre est un pointeur
vers une variable de la même structure.
\begin{itemize}
\item Données éparpillées en mémoire reliées par des pointeurs.
\end{itemize}
\end{block}

\begin{block}{Liste chaînée}
Un membre pointe vers la variable suivante de la liste.
\end{block}

\begin{block}{List doublement chaînée}
Un membre pointe vers la variable suivante et un autre vers
la variable précédente.
\end{block}

\begin{block}{arbre binaire}
Deux pointeurs pointent vers deux membres suivants (les fils).
\end{block}

\end{frame}

\item Supression d'un maillon

\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Création d'un maillon}
\begin{codeblock}{Exemple de modèle de liste}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}
\begin{codeC}
struct liste{
 int val;
 struct liste *suivant;
};

\end{codeC}
\vspace{-.3cm}
\end{codeblock}

\begin{codeblock}{Fonction creation}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}
\begin{codeC}
struct liste * creation() {
 struct liste * nouveau ;
 nouveau = (struct liste *) malloc (sizeof(struct liste));
 nouveau -> val = 0 ; //par exemple
 nouveau -> suivant = NULL ;
 return(nouveau) ;
\end{codeC}
\vspace{-.3cm}
\end{codeblock}

\end{frame}

\begin{frame}[fragile]
\frametitle{Supression}
Supprime l'élément dont la valeur est égale à \Verb|val|.
\begin{codeblock}{Fonction creation}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}
\begin{codeC}
struct liste *supprime(int val, struct liste *tete) {
 struct liste *maillon = tete ;
 struct liste *prec = tete ;
 while (maillon != NULL && maillon -> valeur != val) {
  prec = maillon ;
  maillon = maillon -> suivant ;
 }

 if (maillon != NULL) {
  if (maillon==tete)
   tete = prec -> suivant;
  else 
   prec -> suivant = maillon -> suivant;
  free(maillon);
 }
}
\end{codeC}
\vspace{-.3cm}
\end{codeblock}

\end{frame}

\begin{frame}[fragile]
\frametitle{Autre structure récursive : l'arbre}
\begin{block}{Vocabulaire}
\begin{description}
\item[Noeud de l'arbre] : Elément de base de l'arbre qui contient l'information ;
\item[Fils d'un noeud N] : Noeud pointé par le noeud N ;
\item[Père d'un noeud F] : Noeud qui pointe sur F ;
\item[Racine de l'arbre] : Noeud (unique) qui n'a pas de père ;
\item[Feuille de l'arbre] : Noeud qui n'a aucun fils (pointeurs = \Verb|NULL|).
\end{description}

\end{block}

\begin{frame}[fragile]
\frametitle{Deux types d'arbres}
\begin{block}{Arbre binaire}
Arbre dont chaque noeud a au plus deux fils
(qu'on désigne fils gauche et fils droit)
\end{block}

\begin{block}{Arbre n-aire}
Arbre dont chaque noeud a un nombre indéterminé de fils.
Les fils ont donc une structure de liste chaînée.
\end{block}

\begin{codeblock}{Modèle de structure d'un arbre n-aire}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}

struct arbre {
  int valeur ;
  struct arbre *fils ;
  struct arbre *frere;
}
\end{codeC}
\vspace{-.3cm}
\end{codeblock}
