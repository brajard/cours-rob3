\begin{frame}
\frametitle{L'algorithme minimax}
\begin{itemize}
\setlength\itemsep{1em}
\item Algorithme issu de la théorie des jeux
\item Il s'applique aux jeux à 2 joueurs, à  somme nulle
(1 gagnant/1perdant ou match nul) et à information complète.
\end{itemize}

\begin{block}{Principe}
Passer en revu un nombre limité de coups
et les évaluer en fonction du bénéfice pour le joueur
et sont adversaire
\end{block}

\end{frame}

\begin{overlayarea}{\textwidth}{5cm}

\begin{onlyenv}<1-4|handout:1>
Position initiale donnée\\
Question : quel coup jouer ?
\begin{enumerate}
\item<2-> On explore tous les coups possibles
\item<3-> Pour chaque coup, on explore les réponses possibles de l'adversaire
\item<4-> On continue jusqu'à une profondeur fixée.
\end{enumerate}
\end{onlyenv}

\begin{onlyenv}<5-8|handout:2>
\begin{enumerate}
\item<5-> On évalue les positions des noeuds terminaux (feuilles)
\item<6-> On évalue les noeuds parents
\begin{itemize}
\item<6-> Si le noeud est  \textcolor{red}{joueur} : \texttt{eval(noeud) = max (fils)}
\item<6-> Si le noeud est \textcolor{green}{adversaire} : \texttt{eval(noeud) = min (fils)}
\end{itemize}
\item<7-> Lorsqu'on arrive à la racine, on choisit le coup qui amène au fils maximum
\end{enumerate}
\end{onlyenv}

\begin{onlyenv}<9|handout:3>
\begin{enumerate}
\item Evaluation des feuilles
 \begin{itemize}
\item Si le noeud est  \textcolor{red}{joueur} : \texttt{eval(feuille) = f()}
\item Si le noeud est \textcolor{green}{adversaire} : \texttt{eval(noeud) = -f()}
\end{itemize}
\item Evaluation des noeuds : \texttt{eval(noeud) = max (-eval(fils))}
\item Lorsqu'on arrive à la racine, on choisit le coup qui amène au fils maximum
\end{enumerate}
\begin{alertblock}{}
La fonction d'évaluation doit être symétrique
\end{alertblock}
\end{onlyenv}
