\begin{frame}
\frametitle{Qu'est-ce qu'un bug ? (ou bogue)}
\begin{block}{Définition}
Un bug (ou en français un bogue) est un défaut dans le 
programme informatique du à une erreur de syntaxe ou à
\red{une erreur de conception du programme}
\end{block}

Quelques bugs informatiques célèbres :
\begin{itemize}
\item \href{https://youtu.be/gp_D8r-2hwk}{L'explosion en vol d'Ariane 5 lors du premier tir}
\item Le bug de l'an 2000
\item 99\% de vos futurs programmes.
\end{itemize}

\begin{alertblock}{} 
Il est donc indispensable de déboguer ses programmes.
\end{alertblock}
\end{frame}

\begin{frame}
\frametitle{Déboguage - Les 4 étapes}
\begin{enumerate}
\setlength\itemsep{1.5em}
\item Détecter le bug
\item Localiser le bug
\item Expliquer le bug
\item Réparer le bug
\end{enumerate}

\end{frame}

\begin{frame}
\frametitle{Deux types de bug}
\framesubtitle{1. Le plus simple : le bug à la compilation}
\begin{itemize}
\setlength\itemsep{1.5em}
\item Erreur de syntaxe
\item Variable non déclarée
\item Crochet non fermé
\item Point virgule oublié
\item Erreur de type
\item ...
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Deux types de bug}
\framesubtitle{2. Le plus compliqué : le bug à l'exécution}
\begin{itemize}
\setlength\itemsep{1.5em}
\item Erreur dans la gestion de la mémoire (\Verb|Segmentation fault|)
\item Erreur dans les bornes des boucles(\Verb|for| ou \Verb|while|)
ou dans les tests logiques (\Verb|if|)
\item Aglorithme incorrect (problème de modélisation)
\item ...
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Commandes à utiliser}
\begin{itemize}
\setlength\itemsep{1em}

\item Pour les cas difficiles, 
utiliser un logiciel de déboguage (comme gdb). 
Voir section~\ref{secgdb}.
\item Pour les cas plus simples,
déboguer "à la main".
\begin{itemize}
\item Ajouter des commentaires pour isoler les parties problématiques
du programme.
\item Utiliser \Verb|printf| pour afficher les variables à différentes
étapes.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Le débogueur \Verb|gdb|}

\begin{block} {}
\Verb|gdb| est l'acronyme de "GNU DeBugger".
Il s'agit d'un logicie libre installé sur tout 
système linux permettant de déboguer vos programmes.
\end{block}

\begin{itemize}
\item Il ne nécessite pas de modifier le code
\item Les premières utilisations sont un peu lourde à mettre en place.
\end{itemize}

\end{frame}

\begin{frame}[fragile]
\frametitle{Comment utiliser \Verb|gdb|}
\begin{enumerate}
\setlength\itemsep{1em}
\item Au moment de compiler, ajouter l'option \bvrb|-g| comme option
de compilation\\
\begin{termblock}{}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}
\lstset{moredelim=**[is][\color{red}]{@}{@}}
\begin{lstlisting}
>> gcc -g -Wall monprog.c -o prog
\end{lstlisting}
\vspace{-.3cm}
\end{termblock}

\item Ensuite, lancer : \bvrb |gdb £textitµnom_executable§|\\
\begin{termblock}{}
\vspace{-.3cm}
\lstset{escapeinside={§§}}
\lstset{basicstyle=\scriptsize}
\lstset{moredelim=**[is][\color{red}]{@}{@}}
\begin{lstlisting}
>> gdb prog
\end{lstlisting}
\vspace{-.3cm}
\end{termblock}

\item On arrive ensuite dans un environnement ou l'on peut
faire appel à plusieurs commandes de déboguage.
\end{enumerate}

\end{frame}
