\subsection{Introduction}
\begin{frame}
  \frametitle{Pourquoi faire appel à des algorithmes ?}
  Pour automatiser des tâches
  
  Exemples :
  \begin{itemize}
  \item Métier à tisser\\
  \item Méthode de calcul à la main d'une division\\
  \item Recette de cuisine\\
  \item ...\\
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%% 

\begin{frame}
  \frametitle{Qu'est-ce qu'un algorithme ?}
  \begin{alertblock}{Définition}
    Un algorithme est un ensemble 
    ordonné d'instructions simples
    permettant de résoudre un problème.
  \end{alertblock}
\end{frame}

%%%%%%%%%%%%%%%%%% 

\begin{frame}
\frametitle{Remarques}
  \begin{block}{Un algorithme nécessite :}
    \begin{itemize}
    \item Des objets sur lesquels travailler,\\
    \item Un langage non ambigu,\\
    \item Des spécifications (description de l'algorithme).\\
    \end{itemize}
  \end{block}
  
  \begin{block}{}
    Il n'existe généralement pas un unique algorithme pour traiter un problème.
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Construction d'un algorithme}
\begin{block}{Pour chaque problème, il vous est demandé de définir clairement :}
\begin{itemize}
\item Les éventuelles \red{données d'entrée du problème} en précisant leurs types et leur rôle,\\
\item les éventuelles \red{données de sortie du problème} en précisant leurs types, \\
\item les différentes \red{instructions} permettant d'obtenir les données de sorties à partir
des données d'entrée.\\
\end{itemize}
\end{block}
\end{frame}

\begin{frame}[fragile]
\frametitle{Exemple}
Algorithme qui détermine le prix d'entrée dans un musée (les mineurs payent moitié prix)

\begin{block}{}
\begin{algorithmic}[0]
\STATE{\textbf{Selon} \red{le cas}}
\STATE \hspace{0.8em} Cas 1 : Action 1 
\STATE \hspace{0.8em} Cas 2 : Action 2 
\STATE \hspace{0.8em} ... 
\STATE \hspace{0.8em} Cas n : Action n
\STATE \hspace{0.8em} Défaut : Action par défaut
\STATE{\textbf{Fin Selon}}
\end{algorithmic}
\end{block}

\begin{exampleblock}{Un exemple}
\begin{algorithmic}[0]
\STATE{\textbf{Selon} \red{valeur de age}}
\STATE \hspace{0.8em} 0..6 : tarif $\leftarrow$ 0
\STATE \hspace{0.8em} 7..18 : tarif $\leftarrow$ 4
\STATE \hspace{0.8em} Défaut : tarif $\leftarrow$ 8
\STATE{\textbf{Fin Selon}}
\end{algorithmic}
\end{exampleblock}

\WHILE{\red{condition}}
\STATE Actions
\ENDWHILE
\end{algorithmic}

Les actions peuvent ne \textbf{pas} être exécutées
si la condition est fausse au départ.

\begin{frame}
\frametitle{Jeu d'essais}
\begin{itemize}
\item Faire fonctionner l'algorithme avec des valeurs.
\item Traiter tous les cas possibles.
\item En particulier les cas pouvant nécessaiter à traitement spécifique.
\end{itemize}

\begin{algorithmic}[1]
\REQUIRE{X (entier)}\\
\COMMENT{nombre dont on veut calculer la valeur absolue}

\COMMENT{valeur absolue de X}
\end{algorithmic}

\ENDIF
\RETURN{ Val\_abs}
\end{algorithmic}
\end{exampleblock}

\begin{frame}
\frametitle{Exemple 1}
Un colis est conforme si ses trois dimensions
($L \times l \times h$ avec $h \leq l \leq L$) vérifient
les conditions suivantes : 
\begin{itemize}
\item Un rectangle de 10cm $\times$ 7cm doit pouvoir s'inscrire
dans une face du colis,\\
\item $L + l + h \leq 100$ cm,\\
\item $L \leq 60 cm.$\\
\end{itemize}

\begin{block}{Exercice}
Ecrire l'algorithme permettant de savoir si un colis est conforme ou non.
\end{block}
\end{frame}

\ifdefined\correction
\begin{frame}[fragile]
\frametitle{Exemple 1 -corrigé}
\begin{block}{}
\begin{algorithmic}[0]
\REQUIRE{~\\L (réel),l (réel),h (réel)}
\COMMENT{longeur,largeur,hauteur}
\ENSURE{Pas de sortie}

\IF {($L>10$ \textbf{et} $l>7$)\\ 
\textbf{et} $L+l+h \leq 100$\\
\textbf{et} $L \leq 60$
}
\PRINT "Colis conforme"
\ENDIF
\end{algorithmic}
\end{block}
\end{frame}
\fi

\begin{frame}[fragile]
\frametitle{Exemple 2}
\framesubtitle{Que fait cet algo ?}
\begin{algorithmic}[0]
\REQUIRE{~\\X, tableau à 2 dimension de n$\times$n réels,\\
v, vecteur de n réels}
\ENSURE{s, vecteur de n réels}
\FOR{i variant de 1 à n}
\STATE s(i) $\leftarrow$ 0
\FOR {j variant de 1 à n}
\STATE s(i) $\leftarrow$ s(i) + X(i,j).v(j)
\ENDFOR
\ENDFOR
\RETURN{s}
\end{algorithmic}
\end{frame}

\ifdefined\correction
\begin{frame}
\frametitle{Corrigé}
\begin{block}{Multiplication Matrice-vecteur}
\begin{equation*}
s = X.v
\end{equation*}

\begin{equation*}
s_i = \sum_{j=1}^n X_{ij}.v_j
\end{equation*}


\end{block}
\end{frame}
\fi
