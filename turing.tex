\begin{frame}
\frametitle{Un peu d'histoire...}
\begin{description}
%[font=\color{red}]
\item[1900 : ]Problèmes d'Hilbert (2ème problème : Est-ce que l'arithmétique est cohérente ?)\\
\item[1931 : ]Gödel démontre que l'arithmétique est incomplète. Cela est lié au problème de la décision.\\
\item[1936 : ]Introduction de la machine de Turing par Alan Turing à l'âge de 24 ans pour montrer
que certais problèmes ne peuvent pas être résolus par un algorithme (c'est à dire par la machine de Turing).\\
\item[1942 : ]Participation au décodage d'Enigma.\\
\item[1946 : ]Création de l'ENIAC (Electronic Numerical Integrator and Computer).\\
\item[1954 : ]Mort de Turing.\includegraphics[scale=0.03]{./fig/apple.jpg}\\
\end{description}
\end{frame}

\begin{frame}
\frametitle{Fonctionnement d'une machine de Turing}
\begin{columns}

\column{.4\textwidth}

\begin{enumerate}
\footnotesize
\item<1-|alert@1>La tête de lecture lit un symbole $s$. \tikz[remember picture,baseline=-.5ex] \coordinate(it1);
\item<2-|alert@2>En fonction de $s$ et de son état, la tête écrit un symbole $s'$ \tikz[remember picture,baseline=-.5ex] \coordinate(it2);
\item<3-|alert@3>En fonction de son état la tête se déplace d'un cran.  \tikz[remember picture,baseline=-.5ex] \coordinate(it3);
\item<4-|alert@4>La machine change d'état. \tikz[remember picture,baseline=-.5ex] \coordinate(it4);
\end{enumerate}
